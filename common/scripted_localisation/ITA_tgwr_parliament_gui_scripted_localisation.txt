# Gets the loc to be displayed based on what the current v (value) of the element in the array is (the value corresponds to the party id set in the scripted effect).
defined_text = {
	name = ITA_GetParliamentPartyName
	text = {
		trigger = {
			check_variable = { v = 1 }
		}
		localization_key = ITA_party_1_text
	}
	text = {
		localization_key = "Error"
	}
}

defined_text = {
	name = ITA_GetParliamentPartyDescription
	text = {
		trigger = {
			check_variable = { v = 1 }
		}
		localization_key = ITA_party_1_text_desc
	}
	text = {
		localization_key = "Error"
	}
}

# Since index starts as 0, i can get the current seat number by jut adding it by one and then displaying it.
defined_text = {
	name = ITA_GetParliamentSeatName
	text = {
		trigger = {
			set_temp_variable = { temp_seat = ITA_parliament_main_index }
			add_to_temp_variable = { temp_seat = 1 }
			check_variable = { temp_seat > 0 }
		}
		localization_key = ITA_parliament_seat_text
	}
}
