country_event = { #Supreme Council of the IRB meet #A meeting in ireland
	id = ww1_easter_uprising.1
	title = ww1_easter_uprising.1.t
	desc = ww1_easter_uprising.1.d
	picture = find_one

	fire_only_once = yes

	trigger = {
		#Checks for ENG or GER
		OR = {
			tag = ENG
			tag = GER
		}
		#Checking the ENG controls Ireland
		ENG = {
			owns_state = 113
			owns_state = 135
			owns_state = 134
			owns_state = 119 #Northern Ireland will probs be changed, update this
			#Checking War
			has_war_with = GER
			#Checks war length
			war_length_with = {
				tag = GER
				months = 1
			}
		}
	}

	option = {
		name = ww1_easter_uprising.1.a
		activate_mission = ww1_easter_uprising_mission
		set_variable = {
			var = easter_uprising_british_awareness
			value = 1.0
		}
		clamp_variable = {
			var = easter_uprising_british_awareness
			min = 0
			max = 1.0
		}
		set_variable = {
			var = easter_uprising_irish_organization
			value = 0.15
		}
		clamp_variable = {
			var = easter_uprising_irish_organization
			min = 0
			max = 1.0
		}
		set_variable = {
			var = easter_uprising_german_support
			value = 0
		}
		clamp_variable = {
			var = easter_uprising_german_support
			min = 0
			max = 1.0
		}
		total_readiness_calulation = yes
		hidden_effect = {
			ENG = {
				activate_mission = ww1_rebbelion_development_mission
			}
			GER = {
				country_event = {
					id = ww1_easter_uprising.3
					days = 20
				}
			}
		}
	}
}

country_event = { #Germany gets the initial choice to support Ireland or not
	id = ww1_easter_uprising.4
	title = ww1_easter_uprising.4.t
	desc = ww1_easter_uprising.4.d
	picture = fine_one

	fire_only_once = yes
	is_triggered_only = yes

	option = {
		name = ww1_easter_uprising.4.a #Historical Choice, moderate support
		#Removes guns from Germany's stockpile
		add_equipment_to_stockpile = {
			type = infantry_equipment
			amount = -2000
		}
		#Sets a variable that will be used later to assess how many guns IRE should get
		set_variable = {
			var = easter_uprising_german_guns
			value = 2000
		}
		#Increases German Support variable
		custom_effect_tooltip = IRE_increase_german_support_1
		add_to_variable = {
			var = easter_uprising_german_support
			value = 0.15
		}

			var = easter_uprising_german_support
			min = 0
			max = 1.0
		}
		total_readiness_calulation = yes

		set_global_flag = GER_supported_easter_uprising

		}

		}

		name = ww1_easter_uprising.4.c #No support
		custom_effect_tooltip = IRE_germans_refuse_easter_uprising
		set_global_flag = GER_refused_to_help_easter_uprising
