RUS_prolong_the_vote = {
	custom_effect_tooltip = RUS_prolong_the_vote_tt
	hidden_effect = {
		if = {
			limit = {
				has_decision = RUS_finnish_autonomy_law_sc
			}
			add_days_remove  = {
				decision = RUS_finnish_autonomy_law_sc
				days = 30
			}
		}
		else_if = {
			limit = {
			}
		}
	}
}

RUS_socialist_law = {
	if = {
		limit = {
			check_variable = { state_council_dominant_faction < 26 }
		}
		hidden_effect = {
			set_temp_variable = { RUS_add_law_acceptance_var = 80 }
			RUS_add_law_acceptance = yes
		}
		custom_effect_tooltip = RUS_socialist_law_tt
	}
	else_if = {
		limit = {
			check_variable = { state_council_dominant_faction > 25 }
			check_variable = { state_council_dominant_faction < 51 }
		}
		hidden_effect = {
			set_temp_variable = { RUS_add_law_acceptance_var = 40 }
			RUS_add_law_acceptance = yes
		}
		custom_effect_tooltip = RUS_socialist_law_tt
	}
	else_if = {
		limit = {
			check_variable = { state_council_dominant_faction > 50 }
			check_variable = { state_council_dominant_faction < 76 }
		}
		hidden_effect = {
			set_temp_variable = { RUS_add_law_acceptance_var = 0 }
			RUS_add_law_acceptance = yes
		}
		custom_effect_tooltip = RUS_socialist_law_tt
	}
	else_if = {
		limit = {
			check_variable = { state_council_dominant_faction > 75 }
		}
		hidden_effect = {
			set_temp_variable = { RUS_add_law_acceptance_var = 0 }
			RUS_add_law_acceptance = yes
		}
		custom_effect_tooltip = RUS_socialist_law_tt
	}
}
