# TREATY OF TRIANON
#####################
country_event = {
	id = ww1_hungary.3
	title = ww1_hungary.3.t
	desc = ww1_hungary.3.d
	picture = GFX_report_event_hungary_southern_slovakia
	fire_only_once = yes
	is_triggered_only = yes
	option = {
		name = ww1_hungary.3.a
		add_ideas = HUN_treaty_of_triannon
		add_ideas = idea_army_restrictions
		# Reparations
		add_timed_idea = {
			idea = GER_idea_reparations
			days = 365
		}
		# Territories
		every_country = {
			limit = {
				has_country_flag = hungarian_romanian_war_winner
			}
			every_state = {
				limit = {
					is_core_of = PREV
					is_owned_by = HUN
				}
				PREV = {
					transfer_state = PREV
				}
			}
		}
		# Burgenland
		if = {
			limit = {
				owns_state = 147
			}
			random_other_country = {
				limit = {
					owns_state = 4
				}
				transfer_state = 147
			}
		}
		# Prekmurje
		if = {
			limit = {
				owns_state = 875
			}
			random_other_country = {
				limit = {
					owns_state = 102
				}
				transfer_state = 875
			}
		}
		# Timisoara
		if = {
			limit = {
				OR = {
					owns_state = 82
					FRA = { owns_state = 82 }
				}
			}
			random_other_country = {
				limit = {
					owns_state = 84
				}
				transfer_state = 82
			}
		}
		# Remove claims
		every_state = {
			limit = {
				OR = {
					is_claimed_by = HUN
					is_core_of = HUN
				}
				NOT = {
					is_owned_by = HUN
				}
			}
			remove_claim_by = HUN
			remove_core_of = HUN
		}
		# Hungarian national cores
		every_state = {
			limit = {
				OR = {
					state = 45
					state = 83
					state = 147
					state = 664
					state = 778
					state = 875
				}
			}
			add_core_of = HUN
		}
	}
}
