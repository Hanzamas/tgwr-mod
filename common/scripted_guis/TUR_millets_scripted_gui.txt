scripted_gui = {
	ottoman_armenian_millet_decision_ui = {
		context_type = decision_category
		window_name = "ottoman_armenian_millet_gui_window"
		triggers = {
			TUR_armenian_population_icon_visible = { NOT = { has_country_flag = TUR_armenian_millet_state_selected } }
			TUR_armenian_state_icon_visible = { has_country_flag = TUR_armenian_millet_state_selected }
		}
		effects = {
			TUR_millet_mode_frame_click = {
				if = {
					limit = { has_country_flag = TUR_armenian_millet_state_selected }
					clr_country_flag = TUR_armenian_millet_state_selected
					set_variable = { TUR_millet_mode_bar = 1 }
				}
				else = {
					set_country_flag = TUR_armenian_millet_state_selected
					set_variable = { TUR_millet_mode_bar = 2 }
				}
			}
		}
		properties = { TUR_millet_mode_bar = { frame = TUR_millet_mode_bar } }
	}
}
