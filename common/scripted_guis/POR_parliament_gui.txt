scripted_gui = {
	parliament_gui = {
		context_type = decision_category
		window_name = "POR_parliament_gui_container"

		dynamic_lists = {
			POR_parliament_main_grid = {
				array = POR_parliament_main_array
				index = POR_parliament_main_index
				change_scope = no
				entry_container = POR_parliament_main_entry
			}		
			POR_parliament_semicircle_main_grid = {
				array = POR_parliament_main_array
				index = POR_parliament_main_index
				change_scope = no
				entry_container = POR_parliament_semicircle_main_entry
			}
			POR_parliament_member_grid = {
				array = POR_parliament_member_array
				index = POR_parliament_member_index
				change_scope = no
				entry_container = POR_parliament_member_entry
			}
		}

		effects = {
			parliament_modify_button_control_click = {
				set_temp_variable = { temp_member = v }
				set_temp_variable = { temp_seats = 5 }
				modify_parliament_seat = yes
			}
			parliament_modify_button_shift_click = {
				set_temp_variable = { temp_member = v }
				set_temp_variable = { temp_seats = -5 }
				modify_parliament_seat = yes
			}
		}

		properties = {
			POR_parliament_main_icon = {
				frame = v
			}
			POR_parliament_semicircle_main_icon = {
				frame = v
				x = parliament_x_array^POR_parliament_main_index
				y = parliament_y_array^POR_parliament_main_index
			}
			POR_parliament_member_icon = {
				frame = v
			}
		}
	}
}
