add_namespace = ww1_iraq

# Kingdom of Iraq
country_event = {
	id = ww1_iraq.1
	title = ww1_iraq.1.t
	desc = ww1_iraq.1.d
	picture = GFX_report_event_tur_turkish_soldiers
	fire_only_once = yes
	mean_time_to_happen = {
		days = 23
	}
	trigger = {
		tag = ENG
		owns_state = 291
		NOT = {
			country_exists = SYR
			country_exists = IRQ
		}
		date > 1921.8.1
		has_global_flag = interwar_iraqi_revolt
	}
	option = {
		name = ww1_iraq.1.a
		set_global_flag = kingdom_of_iraq
		IRQ = {
			hidden_effect = {
				SYR = {
					SYR_faisal_i = {
						set_nationality = IRQ
					}
				}
			}
			IRQ_faisal_i = {
				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = autocracy
					}
				}
			}
		}
	}
	option = {
		name = ww1_iraq.1.b
		every_owned_state = {
			limit = {
				is_core_of = IRQ
			}
			transfer_state_to = IRQ
		}
		declare_war_on = {
			target = IRQ
			type = annex_everything
		}
		ai_chance = {
			factor = 0
		}
	}
}

# Iraqi Independence
country_event = {
	id = ww1_iraq.2
	title = ww1_iraq.2.t
	desc = ww1_iraq.2.d
	picture = GFX_report_event_tur_turkish_soldiers
	fire_only_once = yes
	mean_time_to_happen = {
		days = 3
	}
	trigger = {
		tag = ENG
		owns_state = 291
		NOT = {
			country_exists = IRQ
		}
		has_global_flag = kingdom_of_iraq
		date > 1932.8.1
	}
	option = {
		name = ww1_iraq.2.a
		release = IRQ

	}
	option = {
		name = ww1_iraq.1.b
		every_owned_state = {
			limit = {
				is_core_of = IRQ
			}
			transfer_state_to = IRQ
		}
		declare_war_on = {
			target = IRQ
			type = annex_everything
		}
		ai_chance = {
			factor = 0
		}
	}
}
