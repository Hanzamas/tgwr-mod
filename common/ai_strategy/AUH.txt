	allowed = {
		original_tag = AUH	
	}

	enable = {
		tag = AUH
		has_war_with = ITA
		OR = {
			stockpile_ratio = {	## 5% of the armies equipment must be in reserves
				archetype = infantry_equipment
				ratio < 0.050
			}
		}
		NOT = {
			strength_ratio = {
				tag = ITA
				ratio > 0.9
			}
		}
	}
	abort_when_not_enabled = yes
	
	ai_strategy = {
		type = front_control
		tag = ITA
		ratio = 0.5
		
		priority = 10
		ordertype = front
		execution_type = careful
		manual_attack = no
		execute_order = no
	}

	ai_strategy = {
		type = put_unit_buffers
		
		order_id = 100
		
		# protect northern front
		states = {
			244
		}
		ratio = 0.20
		subtract_invasions_from_need = yes
	}

}
