	desc = ottoman_syria.4.d
	is_triggered_only = yes
	option = {
		name = ottoman_syria.4.a
		effect_tooltip = {
			FROM = {
				send_equipment = {
					equipment = infantry_equipment
					amount = 500
					target = ROOT
				}
			}
		}
		if = {
			limit = {
				check_variable = { TUR_o4a_army_attack_factor < 0.05 }
				check_variable = { TUR_o4a_army_defence_factor < 0.05 }
			}

			add_to_variable = { TUR_o4a_army_defence_factor = 0.01 }
		}
		ai_chance = { base = 100 }
	}
}
