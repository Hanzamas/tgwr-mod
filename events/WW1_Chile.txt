#Pedro Montt dies
country_event = {
	id = ww1_chile.1
	title = ww1_chile.1.t
	desc = ww1_chile.1.d
	picture = GFX_report_event_generic_funeral
	trigger = {
		original_tag = CHL
		date > 1910.8.1
		has_country_leader = {
			character = CHL_pedro_montt
			ruling_only = yes
		}
	}
	immediate = {
		hidden_effect = {
			retire_character = CHL_pedro_montt
			promote_character = CHL_elias_fernandez_albano
			country_event = {
				id = ww1_chile.2
				days = 21
			}
		}
	}
	fire_only_once = yes
	mean_time_to_happen = {
		days = 16
	}
	option = {
		name = ww1_chile.1.a
		effect_tooltip = {
			retire_character = CHL_pedro_montt
			promote_character = CHL_elias_fernandez_albano
		}
	}
}
