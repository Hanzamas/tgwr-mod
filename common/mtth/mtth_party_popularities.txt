neutrality_party_popularity_var = {
	base = 0
	modifier = {
        has_country_flag = neutrality_party_popularity
		neutrality > 0

		set_temp_variable = { neutrality_party_popularity_var_temp = party_popularity@neutrality }
		if = {
			limit = { has_completed_focus = TUR_unity_of_the_party_and_the_state }
			set_temp_variable = { neutrality_party_popularity_var_TUR_unity_of_the_party_and_the_state = party_popularity@neutrality }
			multiply_temp_variable = { neutrality_party_popularity_var_TUR_unity_of_the_party_and_the_state = 0.1 }
			add_to_temp_variable = { neutrality_party_popularity_var_temp = neutrality_party_popularity_var_TUR_unity_of_the_party_and_the_state }
		}
	}
}
