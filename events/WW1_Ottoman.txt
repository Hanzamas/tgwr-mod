#We take Egypt after op- also by UU ;)
country_event = {
	id = ww1_ottoman.16
	title = ww1_ottoman.16.t
	desc = ww1_ottoman.16.d
	picture = GFX_report_event_tur_turkish_soldiers
	trigger = {
		tag = TUR
		has_country_flag = great_war_winner
		OR = {
			has_country_flag = ottomans_hold_cairo_and_alexandria
			has_subject = EGY
		}
		NOT = {
			has_war_with = EGY
		}
	}
	fire_only_once = yes
	option = {
		name = ww1_ottoman.16.a
		hidden_effect = {
			every_state = {
				limit = {
					OWNER = {
						has_country_flag = great_war_loser
					}
					OR = {
						is_core_of = EGY
						is_core_of = SUD
						is_core_of = DAR
					}
				}
				# Take Suez and Ottoman core states
				if = {
					limit = {
						OR = {
							state = 447
							state = 453
							is_core_of = ROOT
						}
					}
					transfer_state_to = ROOT
					else = {
						# The others belongs to the Khedivate
						transfer_state_to = EGY
					}
				}
				# Annex it if Egypt is no longer has any Egyptian core state
				EGY = {
					if = {
						limit = {
							NOT = {
								any_owned_state = {
									is_core_of = PREV
								}
							}
						}
						ROOT = {
							annex_country = {
								target = PREV
							}
						}
					}
				}
			}
			# No longer puppet of the defeated one
			every_other_country = {
				limit = {
					has_subject = EGY
					has_country_flag = great_war_loser
				}
				end_puppet = EGY
			}
			EGY = {
				country_lock_all_division_template = no
				# Pro Ottoman Khedive
				promote_character = EGY_abbas_ii_helmy
				drop_cosmetic_tag = yes
				# Fix Senussian border
				if = {
					limit = {
						country_exists = LBA
					}
					every_owned_state = {
						limit = {
							OR = {
								state = 552
								state = 767
							}
						}
						transfer_state_to = LBA
					}
				}
			}
			# Puppet Khedivate
			if = {
				limit = {
					NOT = { has_subject = EGY }
				}
				set_autonomy = {
					target = EGY
					autonomous_state = autonomy_dominion
					freedom_level = 0.8
				}
				# For the Ottoman socialist path
				if = {
					limit = {
						has_government = communism
					}
					EGY = {
						set_politics = {
							ruling_party = communism
							elections_allowed = no
						}
						set_popularities = {
							communism = 100
						}
					}
					# For the other paths
					else = {
						EGY = {
							set_politics = {
								ruling_party = neutrality
								elections_allowed = no
							}
							set_popularities = {
								democratic = 34
								neutrality = 65
							}
						}
					}
				}
			}
		}
	}
}
