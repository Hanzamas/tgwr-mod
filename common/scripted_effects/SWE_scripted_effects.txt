SWE_increase_emigration = {
	if = {
		limit = { has_idea = SWE_high_emigration_1 }
		swap_ideas = {
			add_idea = SWE_high_emigration_0
			remove_idea = SWE_high_emigration_1
		}
	}
	else_if = {
		limit = { has_idea = SWE_high_emigration_2 }
		swap_ideas = {
			add_idea = SWE_high_emigration_1
			remove_idea = SWE_high_emigration_2
		}
	}
	else_if = {
		limit = { has_idea = SWE_high_emigration_3 }
		swap_ideas = {
			add_idea = SWE_high_emigration_2
			remove_idea = SWE_high_emigration_3
		}
	}
	else_if = {
		limit = { has_idea = SWE_high_emigration_4 }
		swap_ideas = {
			add_idea = SWE_high_emigration_3
			remove_idea = SWE_high_emigration_4
		}
	}
	else_if = {
		limit = { has_idea = SWE_high_emigration_5 }
		swap_ideas = {
			add_idea = SWE_high_emigration_4
			remove_idea = SWE_high_emigration_5
		}
	}
}

SWE_set_kingly_approval = {
	custom_effect_tooltip = replaces_current_gustaf_spirit_with
	hidden_effect = {
		remove_ideas = {
			SWE_gustaf_v
			SWE_gustaf_v_apr
			SWE_gustaf_v_dis
			SWE_gustaf_v_opp
		}
	}
	if = {
		limit = {
			OR = {
				has_government = communism
				has_country_leader = {
					character = SWE_karl_staaff
					ruling_only = yes
				}
			}
		}
		add_ideas = SWE_gustaf_v_opp
	}
	else_if = {
		limit = {
			OR = {
				has_country_leader_ideology = conservatism
				has_country_leader_ideology = liberal_conservatism
				has_country_leader = {
					character = SWE_ernst_trygger
					ruling_only = yes
				}
				has_country_leader = {
					character = SWE_sven_hedin
					ruling_only = yes
				}
			}
		}
		add_ideas = SWE_gustaf_v_apr
	}
	else_if = {
		limit = {
			OR = {
				has_country_leader_ideology = social_liberalism
				has_country_leader_ideology = social_democracy
				has_country_leader_ideology = national_syndicalism
				has_country_leader_ideology = socialism
			}
		}
		add_ideas = SWE_gustaf_v_dis
	}
	else = {
		add_ideas = SWE_gustaf_v
	}
}

SWE_increase_entente_point_amount = {
	if = {
		limit = {
			check_variable = { var = SWE_entente_points value = 1 }
		}
		add_to_variable = {
			var = SWE_entente_points
			value = 1
			tooltip = will_increase_entete_points_tt_alt
		}
	}
	else_if = {
		limit = {
			check_variable = { var = SWE_entente_points value = 1 compare = less_than }
		}
		custom_effect_tooltip = will_increase_entete_points_tt
		set_variable = { var = SWE_entente_points value = 1 }
	}
}
