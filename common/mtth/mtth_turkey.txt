TUR_separatist_sentiments_stability = {
	base = 0
	modifier = {
		set_temp_variable = { TUR_minority_state_variable_stability = 0 }
		all_state = {
			if = {
				limit = { TUR_state_has_minority_mtth = yes }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_armenian_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_greek_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_arab_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_jewish_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_south_slavic_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_albanian_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_kurdish_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_assyrian_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_georgian_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_romanian_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_turkish_population_stability_factor }
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_azerbaijani_population_stability_factor }
			}
			if = {
				limit = { TUR_state_has_autonomy_lebanon_mtth = yes }
				add_to_temp_variable = { TUR_minority_state_variable_stability = 0.02 }
			}
			if = {
				limit = { TUR_state_has_autonomy_mtth = yes }
				add_to_temp_variable = { TUR_minority_state_variable_stability = TUR.TUR_ottoman_autonomy_stability_factor }
			}
			if = {
				limit = { TUR_state_has_semi_autonomy_mtth = yes }
				add_to_temp_variable = { TUR_minority_state_variable_stability = TUR.TUR_ottoman_semi_autonomy_stability_factor }
			}
		}
		add = TUR_minority_state_variable_stability
	}
}

			if = {
				limit = { has_dynamic_modifier = { modifier = TUR_tax_collection } }

			}
		}
		add = TUR_tax_collection_var
	}
