TUR_minorities_stability = {
	base = 0
	modifier = {
		set_temp_variable = { TUR_minority_state_variable_stability = 0 }
		all_controlled_state = {
			if = {
				limit = {
					TUR_state_has_any_minority_modifier = yes
				}
				add_to_temp_variable = { TUR_minority_state_variable_stability = THIS.TUR_ottoman_state_stability_factor }
			}
			if = {
				limit = {
					has_dynamic_modifier = { modifier = TUR_autonomous_vilayet }
				}
				add_to_temp_variable = { TUR_minority_state_variable_stability = TUR.TUR_ottoman_autonomy_stability_factor }
			}
		}

		add = TUR_minority_state_variable_stability
	}
}
