# The following scripted effect does the main job of generating the gui, by adding the seats for each party in a grid.
# This should be called every time you need to update the gui, when you add/subtract seats for example.
# Read the comments if you're interested in how it works
POR_reload_parliament_gui = {
	# Clears what's currently on the gui, if there's anything there.
	clear_array = POR_parliament_main_array

	# The array POR_temp_seat is duplicated from the array POR_parliament_seat_array.
	for_each_loop = {
		array = POR_parliament_seat_array

		add_to_array = { POR_temp_seat = POR_parliament_seat_array^i }
	}

	# It's a suprise tool for later
	set_temp_variable = { temp_index_1 = 0 }
	set_temp_variable = { temp_index_2 = 1 }

	# Set temp_size as the highest index in the array, highest index is the highest value x in temp_array^x.
	set_temp_variable = { temp_size = POR_temp_seat^num }
	subtract_from_temp_variable = { temp_size = 1 }

	# What this does is loop through the POR_temp_seat array, adding the current value to the next value on the list.
	# POR_temp_seat^0 is the first value on the list, POR_temp_seat^1 is the next and so forth.
	# the loop ends when temp_index_1 is equal to the size of the array - 1 (basically the max index).
	while_loop_effect = {
		limit = {
			check_variable = { temp_index_1 < temp_size }
		}
		# The suprise tool starts out as POR_temp_seat^0 and POR_temp_seat^1 and then is incremented every loop , adding the current value to the next value on the list.
		add_to_variable = { POR_temp_seat^temp_index_2 = POR_temp_seat^temp_index_1 }
		add_to_temp_variable = { temp_index_1 = 1 }
		add_to_temp_variable = { temp_index_2 = 1 }
	}

	# resize_array fills the POR_parliament_member_array with the id of the parliament member x amount of times where x is the number of seats the member has.
	# so the array will look like this basically:
	# 1,1,1,1,1,1,1,...(repeated 90 times),2,2,2,2,2,2,2(repeated 35 times),3,3,3.. etc.
	for_loop_effect = {
		start = 0
		end = POR_temp_seat^num

		resize_array = { array = POR_parliament_main_array value = POR_parliament_member_array^v size = POR_temp_seat^v }
	}

	# Clear the POR_temp_seat array used previously since it's not needed anymore.
	clear_array = POR_temp_seat
}

# Bonus code for adding/subtracting seats
modify_parliament_seat = {
	subtract_from_temp_variable = { POR_POR_temp_member = 1 }

	# check to see if result overflows
	set_temp_variable = { temp = POR_parliament_seat_array^POR_POR_temp_member }
	add_to_temp_variable = { temp = POR_temp_seats }

	if = {
		limit = {
			check_variable = { temp > POR_parliament_main_array^num }
		}
		subtract_from_temp_variable = { temp = POR_parliament_main_array^num }
		subtract_from_temp_variable = { POR_temp_seats = temp }
	}
	else_if = {
		limit = {
			check_variable = { temp < 0 }
		}
		multiply_temp_variable = { temp = -1 }
		subtract_from_temp_variable = { POR_temp_seats = temp }
	}

	# adds all parties to temp_array
	for_loop_effect = {
		start = 0
		end = POR_parliament_seat_array^num

		add_to_temp_array = { temp_array = v }
	}

	# remove from list of parties the party to be added/subtracted
	remove_from_temp_array = { array = temp_array value = POR_POR_temp_member }

	# give party new seat value
	add_to_variable = { POR_parliament_seat_array^POR_POR_temp_member = POR_temp_seats }

	# if its add
	if = {
		limit = {
			check_variable = { POR_temp_seats > 0 }
		}
		# loop until seats to be added is 0
		while_loop_effect = {
			limit = {
				check_variable = { POR_temp_seats > 0 }
			}

			# loops through remaining parties, subtracting 1 seat each time
			for_loop_effect = {
				start = 0
				end = temp_array^num

				if = {
					limit = {
						check_variable = { POR_parliament_seat_array^temp_array^v > 0 } # party must have more than 0 seats to be subtracted
						check_variable = { POR_temp_seats > 0 } # seats to be added is bigger than 0 currently
					}
					subtract_from_variable = { POR_parliament_seat_array^temp_array^v = 1 } # subtract from party 1 seat
					subtract_from_temp_variable = { POR_temp_seats = 1 } # subtract from seats to be added
				}
			}
		}
	}
	# if its subtract
	else_if = {
		limit = {
			check_variable = { POR_temp_seats < 0 }
		}
		# loop until seats to be subtracted is 0
		while_loop_effect = {
			limit = {
				check_variable = { POR_temp_seats < 0 }
			}

			# loops through remaining parties, adding 1 seat each time
			for_loop_effect = {
				start = 0
				end = temp_array^num

				if = {
					limit = {
						check_variable = { POR_parliament_seat_array^temp_array^v < POR_parliament_main_array^num } # party must have less than max seats to be added
						check_variable = { POR_temp_seats < 0 } # seats to be subtracted is less than 0 currently
					}
					add_to_variable = { POR_parliament_seat_array^temp_array^v = 1 } # add to party 1 seat
					add_to_temp_variable = { POR_temp_seats = 1 } # add to seats to be subtracted
				}
			}
		}
	}
	# do nothing if 0

	# Modify what shows up in the gui
	POR_reload_parliament_gui = yes
}

POR_recalculate_parliament_alliance = {
    set_variable = { var = coalition_alliance value = 0 } 
	if = {
        limit = {
            has_country_flag = progressive_party_allied
        }
        add_to_variable = { var = coalition_alliance value = progressive_party }
    }
    if = {
        limit = {
            has_country_flag = progressive_dissidents_allied
        }
        add_to_variable = { var = coalition_alliance value = progressive_dissidents }
    }
    if = {
        limit = {
            has_country_flag = portuguese_republican_party_allied
        }
        add_to_variable = { var = coalition_alliance value = portuguese_republican_party }
    }
    if = {
        limit = {
            has_country_flag = independents_allied
        }
        add_to_variable = { var = coalition_alliance value = independents }
    }	
	set_variable = {
		var = coalition_alliance_percent
		value = coalition_alliance
	}
	multiply_variable = {
		coalition_alliance_percent = 100
	}
	divide_variable = {
		coalition_alliance_percent = seatNumbers
	}
    POR_reload_parliament_gui = yes
}
