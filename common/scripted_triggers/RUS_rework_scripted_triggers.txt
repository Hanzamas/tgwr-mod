reactionary_prime_minister = {
	custom_trigger_tooltip = {
		tooltip = reactionary_prime_minister_scripted_trigger_tt
		RUS = {
			any_character = {
				has_trait = prime_minister
				has_trait = reactionary
			}
		}
	}
}

	custom_trigger_tooltip = {
		tooltip = not_reactionary_prime_minister_scripted_trigger_tt
		RUS = {
			any_character = {
				has_trait = prime_minister
				NOT = {
					has_trait = reactionary
				}
			}
		}
	}
}
