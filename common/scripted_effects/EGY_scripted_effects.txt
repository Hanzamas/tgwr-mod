
EGY_decrease_feudalism_idea = {
	custom_effect_tooltip = EGY_reform_tt
	hidden_effect = {
		if = {
			limit = { has_idea = EGY_elitist_grip_1 }
			swap_ideas = {
				remove_idea = EGY_elitist_grip_1
				add_idea = EGY_elitist_grip_2
			}
		}
		else_if = {
			limit = { has_idea = EGY_elitist_grip_2 }
			swap_ideas = {
				remove_idea = EGY_elitist_grip_2
				add_idea = EGY_elitist_grip_3
			}
		}
		else_if = {
			limit = { has_idea = EGY_elitist_grip_3 }
			swap_ideas = {
				remove_idea = EGY_elitist_grip_3
				add_idea = EGY_elitist_grip_4
			}
		}
		else_if = {
			limit = { has_idea = EGY_elitist_grip_4 }
			swap_ideas = {
				remove_idea = EGY_elitist_grip_4
				add_idea = EGY_elitist_grip_5
			}
		}
	}
}
