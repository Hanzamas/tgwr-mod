country_event = {
	id = ww1_rcw.6
	title = ww1_rcw.6.t
	desc = ww1_rcw.6.d
	picture = GFX_report_event_generic_military_parade
	is_triggered_only = yes
	trigger = {
		KAZ = {
			exists = no
		}
		any_owned_state = {
			NOT = {
				state = 590
				state = 587
			}
			is_core_of = KAZ
		}
	}
	immediate = {
		hidden_effect = {
			every_state = {
				limit = {
					NOT = {
						state = 590
						state = 587
					}
					is_core_of = KAZ
					is_owned_by = RUS
				}
				KAZ = {
					transfer_state = PREV
				}
			}
			KAZ = {
				set_cosmetic_tag = KAZ_alash
				set_province_name = {
					id = 4661
					name = ALASH_PROV
				}
				hidden_effect = {
					load_oob = "KAZ_1918"
					inherit_technology = RUS
				}
			}
			if = {
				limit = {
					OR = {
						has_dlc = "Together for Victory"
						has_dlc = "Man the Guns"
					}
				}
				RUS = {
					set_autonomy = {
						target = KAZ
