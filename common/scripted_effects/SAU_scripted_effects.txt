increase_SAU_Urban = {
	custom_effect_tooltip = SAU_Urbans_Happy
	if = {
		limit = {
			check_variable = {
				SAU_Urban_invetments < 0.4
			}
		}
		add_to_variable = {
			SAU_Urban_invetments = -0.05
		} 
	}
	if = {
		limit = {
			check_variable = {
				SAU_Urban_political_power < 0.4
			}
		}
		add_to_variable = {

		} 
	}
	if = {

		} 
	}
}
